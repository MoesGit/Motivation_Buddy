var ss = SpreadsheetApp.openByUrl('https://docs.google.com/spreadsheets/d/10_N4nuXHupbONZOH2cjgqTWPJeQJqCyCI_WutYVLkno/edit#gid=0');
var sheet = ss.getSheetByName('RemotePartner_Debo');

function doPost(e) {
  var val = e.parameter.value;
  
  if (e.parameter.value !== undefined){
    var range = sheet.getRange('A2');
    range.setValue(val);
  }
}

function doGet(e){
  var read = e.parameter.read;

  if (read !== undefined){
    return ContentService.createTextOutput(sheet.getRange('H4').getValue());
  }
}
